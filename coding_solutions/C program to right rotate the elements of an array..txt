#include<stdio.h> 
{ 
int arr[]={1,2,3,4,5}; 
int len=sizeof(arr)/sizeof(arr[0]); 
int n=3; 
int i; 
printf(“Original array:”); 
for(i=0;i<len;i++) 
{ 
 printf(“%d”,arr[i]); 
} 
for(i=0;i<n;i++) 
{ 
 int j,last; 
 last=arr[len-1]; 
 for(j=len-1;j>0;j--) 
 { 
  arr[j]=arr[j-1]; 
 } 
 arr[0]=last; 
} 
printf(“\n”); 
printf(“Array after right rotation:\n”); 
for(i=0;i<len;i++) 
{ 
 printf(“%d”,arr[i]); 
} 
return 0; 
}  
OUTPUT: 
Original array: 
1  2  3  4  5 
Array after right rotation: 
3  4  5  1  2
